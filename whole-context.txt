function wt() {

# Usage helper

if [ "$1" == "-h" ] || [ "$1" == "--help" ] || [ -z "$1" ]; then

echo "Usage: wt <context> <project_name> <branch_name> [optional_app1] [optional_app2]..."

echo "Contexts: personal | company"

echo "Examples:"

echo " wt personal project1 feature/mono-repo-fix"

echo " wt company workelate feature/new-api backend hub tasknetic"

return 0

fi



local CONTEXT=$1

local PROJECT_NAME=$2

local BRANCH=$3

shift 3 # Shift arguments to leave only the optional targeted applications ($@)

local TARGET_APPS=("$@")



local MAIN_DIR=""

local WT_ROOT=""

local WORKSPACE_DEST=""



# =========================================================================

# 🛠️ QUICK EDIT CONFIGURATION BLOCK (Modify your paths here)

# =========================================================================

if [ "$CONTEXT" == "personal" ]; then

MAIN_DIR="$HOME/Documents/projects"

WT_ROOT="$HOME/Documents/projects/worktree"

WORKSPACE_DEST="$WT_ROOT" # Saves workspace directly inside the worktree folder

elif [ "$CONTEXT" == "company" ]; then

MAIN_DIR="$HOME/workelate/main"

WT_ROOT="$HOME/workelate/worktree"

WORKSPACE_DEST="$HOME/workelate" # Matches your preference: inside workelate/

else

echo "❌ Error: Invalid context. Use 'personal' or 'company'."

return 1

fi

# =========================================================================



# Clean branch name strings to format safe workspace filenames (e.g., replace / with -)

local SAFE_BRANCH=$(echo "$BRANCH" | sed 's/\//-/g')

local TARGET_WT_DIR="$WT_ROOT/${PROJECT_NAME}-${SAFE_BRANCH}"

mkdir -p "$TARGET_WT_DIR"



# Case A: Mono-repo (No specific target sub-apps defined in command)

if [ ${#TARGET_APPS[@]} -eq 0 ]; then

echo "📦 Mono-repo detected for '$PROJECT_NAME'. Creating single worktree..."


local MONO_SOURCE="$MAIN_DIR/$PROJECT_NAME"

if [ ! -d "$MONO_SOURCE" ] && [ "$CONTEXT" == "personal" ]; then

MONO_SOURCE="$MAIN_DIR/$PROJECT_NAME"

elif [ ! -d "$MONO_SOURCE" ]; then

# Fallback if personal projects are flat in the root directory

MONO_SOURCE="$MAIN_DIR"

fi



cd "$MONO_SOURCE" || { echo "❌ Root source directory not found"; return 1; }

git fetch origin

git worktree add "$TARGET_WT_DIR" -b "$BRANCH"


echo "✨ Launching VS Code for Mono-Repo..."

code "$TARGET_WT_DIR"

return 0

fi



# Case B: Multi-repo Suite (Specific sub-apps defined)

echo "🗂️ Suite configuration detected. Extracting target apps..."


# Start building the JSON structure for the VS Code workspace file

local WORKSPACE_FILE="$WORKSPACE_DEST/${PROJECT_NAME}-${SAFE_BRANCH}.code-workspace"

echo "{" > "$WORKSPACE_FILE"

echo " \"folders\": [" >> "$WORKSPACE_FILE"



local FIRST=true

for APP in "${TARGET_APPS[@]}"; do

local APP_SOURCE="$MAIN_DIR/$APP"

local APP_DEST="$TARGET_WT_DIR/$APP"



if [ ! -d "$APP_SOURCE" ]; then

echo "⚠️ Skipping '$APP': Source directory does not exist at $APP_SOURCE"

continue

fi



echo "⚙️ Branching worktree for: $APP"

cd "$APP_SOURCE" || continue

git fetch origin

git worktree add "$APP_DEST" -b "$BRANCH"



# Append the directory pointer string to the workspace configuration JSON array

if [ "$FIRST" = true ]; then

FIRST=false

else

echo "," >> "$WORKSPACE_FILE"

fi

echo -n " { \"path\": \"$APP_DEST\" }" >> "$WORKSPACE_FILE"

done



# Close out the VS Code workspace JSON layout formatting

echo "" >> "$WORKSPACE_FILE"

echo " ]," >> "$WORKSPACE_FILE"

echo " \"settings\": {}" >> "$WORKSPACE_FILE"

echo "}" >> "$WORKSPACE_FILE"



echo "✅ Saved workspace file to: $WORKSPACE_FILE"

echo "✨ Opening isolated workspace in VS Code..."

code "$WORKSPACE_FILE"

}

PowerShell
function wtf {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true, Position=0)] [string]$Action,
        [Parameter(Position=1)] [string]$Context,
        [Parameter(Position=2)] [string]$Project,
        [Parameter(Position=3)] [string]$Branch,
        [Parameter(ValueFromRemainingArguments=$true)] $Apps
    )

    $ConfigPath = "$env:USERPROFILE\.wtf\config.json"
    if (-not (Test-Path $ConfigPath)) { Write-Host "❌ Config not found at $ConfigPath" -ForegroundColor Red; return }
    $Config = Get-Content $ConfigPath | ConvertFrom-Json -AsHashtable

    # Validate Context & Project
    if (-not $Config.contexts.Contains($Context)) { Write-Host "❌ Context '$Context' not found." -ForegroundColor Red; return }
    $ProjConfig = $Config.contexts[$Context].projects[$Project]
    if ($null -eq $ProjConfig) { Write-Host "❌ Project '$Project' not found." -ForegroundColor Red; return }

    $MainDir = $Config.contexts[$Context].mainDir
    $WtRoot = $Config.contexts[$Context].worktreeDir
    $SafeBranch = $Branch -replace '/', '-'
    $FeatureDir = Join-Path $WtRoot "$Project-$SafeBranch"

    switch ($Action) {
        "create" {
            if (Test-Path $FeatureDir) { Write-Host "⚠️ Worktree suite already exists. Use 'wtf open' or 'wtf add'." -ForegroundColor Yellow; return }
            New-Item -ItemType Directory -Path $FeatureDir | Out-Null
            
            $Meta = @{ context = $Context; project = $Project; branch = $Branch; apps = @() }
            
            foreach ($app in $Apps) {
                if (-not $ProjConfig.apps.Contains($app)) { continue }
                $AppMainPath = Join-Path $MainDir $ProjConfig.apps[$app]
                $AppWtPath = Join-Path $FeatureDir $app
                
                Write-Host "🌱 Branching $app..." -ForegroundColor Cyan
                Set-Location $AppMainPath
                git fetch origin quiet
                git worktree add $AppWtPath -b $Branch

                # The Env Bridge: Copying ignored variables
                foreach ($envFile in @(".env", ".env.local", ".env.development")) {
                    $srcEnv = Join-Path $AppMainPath $envFile
                    if (Test-Path $srcEnv) { Copy-Item $srcEnv -Destination (Join-Path $AppWtPath $envFile) }
                }
                $Meta.apps += $app
            }

            # Generate Agent Plan Markdown
            $PlanPath = Join-Path $FeatureDir "_PLAN.md"
            @"
# Feature: $Branch
**Apps in Scope:** $($Meta.apps -join ', ')

## 🧠 Context / Goal
(Define the root architecture here before diving into Agent sessions)

## 📋 Execution Steps
- [ ] Backend updates
- [ ] Frontend integration
"@ | Out-File $PlanPath -Encoding utf8

            # Build VS Code Workspace (Relative paths)
            $WsFolders = $Meta.apps | ForEach-Object { @{ path = $_ } }
            $Workspace = @{ folders = $WsFolders; settings = @{} }
            $Workspace | ConvertTo-Json -Depth 5 | Out-File (Join-Path $FeatureDir "$SafeBranch.code-workspace")

            # Save Meta State
            $Meta | ConvertTo-Json | Out-File (Join-Path $FeatureDir ".wtf-meta.json")
            
            wtf open $Context $Project $Branch
        }

        "add" {
            if (-not (Test-Path $FeatureDir)) { Write-Host "❌ Feature dir not found." -ForegroundColor Red; return }
            $MetaFile = Join-Path $FeatureDir ".wtf-meta.json"
            $Meta = Get-Content $MetaFile | ConvertFrom-Json
            
            foreach ($app in $Apps) {
                if ($app -in $Meta.apps) { continue }
                $AppMainPath = Join-Path $MainDir $ProjConfig.apps[$app]
                $AppWtPath = Join-Path $FeatureDir $app
                
                Write-Host "➕ Injecting $app into suite..." -ForegroundColor Green
                Set-Location $AppMainPath
                git fetch origin quiet
                git worktree add $AppWtPath -b $Branch
                
                # Copy envs
                foreach ($envFile in @(".env", ".env.local")) {
                    if (Test-Path "$AppMainPath\$envFile") { Copy-Item "$AppMainPath\$envFile" -Destination "$AppWtPath\$envFile" }
                }
                $Meta.apps += $app
            }

            # Update Meta & Workspace
            $Meta | ConvertTo-Json | Out-File $MetaFile
            $WsPath = Join-Path $FeatureDir "$SafeBranch.code-workspace"
            $Workspace = Get-Content $WsPath | ConvertFrom-Json
            $Workspace.folders = $Meta.apps | ForEach-Object { @{ path = $_ } }
            $Workspace | ConvertTo-Json -Depth 5 | Out-File $WsPath

            Write-Host "✅ App added! Workspace updated. Run 'wtf open' again if you need the new terminals." -ForegroundColor Cyan
        }

        "open" {
            $MetaFile = Join-Path $FeatureDir ".wtf-meta.json"
            if (-not (Test-Path $MetaFile)) { Write-Host "❌ Meta file missing in $FeatureDir" -ForegroundColor Red; return }
            $Meta = Get-Content $MetaFile | ConvertFrom-Json

            Write-Host "✨ Launching VS Code..." -ForegroundColor Magenta
            code (Join-Path $FeatureDir "$SafeBranch.code-workspace")

            # ---------------------------------------------------------
            # 1. WINDOW: AGENTS (Split Panes, Only feature apps)
            # ---------------------------------------------------------
            $agentArgs = @("-w", "ag_$SafeBranch")
            $first = $true
            foreach ($app in $Meta.apps) {
                if (-not $first) { $agentArgs += ";"; $agentArgs += "split-pane" } else { $agentArgs += "new-tab"; $first = $false }
                $agentArgs += "-d"; $agentArgs += "$FeatureDir\$app"
                $agentArgs += "--title"; $agentArgs += "🤖 $app (Agent)"
                $agentArgs += "--tabColor"; $agentArgs += "#8A2BE2" # Purple
            }
            Write-Host "🤖 Booting Agent Panes..." -ForegroundColor DarkGray
            & wt.exe $agentArgs

            # ---------------------------------------------------------
            # 2. WINDOW: RUNNERS (Tabs, All apps in project)
            # ---------------------------------------------------------
            $AllApps = $ProjConfig.apps.Keys
            $runArgs = @("-w", "run_$SafeBranch")
            $first = $true
            foreach ($app in $AllApps) {
                if (-not $first) { $runArgs += ";"; $runArgs += "new-tab" } else { $runArgs += "new-tab"; $first = $false }
                
                if ($app -in $Meta.apps) {
                    $runArgs += "-d"; $runArgs += "$FeatureDir\$app"
                    $runArgs += "--title"; $runArgs += "🚀 $app [F]"
                    $runArgs += "--tabColor"; $runArgs += "#00FF7F" # Bright Green for Feature
                } else {
                    $runArgs += "-d"; $runArgs += "$MainDir\$($ProjConfig.apps[$app])"
                    $runArgs += "--title"; $runArgs += "📦 $app [M]"
                    $runArgs += "--tabColor"; $runArgs += "#555555" # Dim Gray for Main
                }
            }
            Write-Host "🏃 Booting Runner Tabs..." -ForegroundColor DarkGray
            & wt.exe $runArgs
        }

        "remove" {
            $MetaFile = Join-Path $FeatureDir ".wtf-meta.json"
            if (-not (Test-Path $MetaFile)) { Write-Host "❌ Meta file not found." -ForegroundColor Red; return }
            $Meta = Get-Content $MetaFile | ConvertFrom-Json

            Write-Host "🛡️ Running Git Safety Checks..." -ForegroundColor Cyan
            foreach ($app in $Meta.apps) {
                $AppWtPath = Join-Path $FeatureDir $app
                Set-Location $AppWtPath
                
                $status = git status --porcelain
                if ($status) {
                    Write-Host "❌ ABORT: Uncommitted changes in '$app'!" -ForegroundColor Red
                    Write-Host $status -ForegroundColor Yellow
                    return
                }
            }

            Write-Host "🧹 Worktrees clean. Executing teardown..." -ForegroundColor Green
            foreach ($app in $Meta.apps) {
                $AppMainPath = Join-Path $MainDir $ProjConfig.apps[$app]
                Set-Location $AppMainPath
                git worktree remove (Join-Path $FeatureDir $app)
                git worktree prune
            }
            
            Set-Location $MainDir
            Remove-Item -Path $FeatureDir -Recurse -Force
            Write-Host "✅ Feature suite destroyed cleanly." -ForegroundColor Green
        }
    }
}
